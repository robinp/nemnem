Returns either a 'ParseError' -- ('Left') or a value of type @a@ ('Right'). -- -- > main = do{ result <- parseFromFile numbers "digits.txt" -- > ; case result of -- > Left err -> print err -- > Right xs -> print (sum xs) -- > } parseFromFile :: Parser a -> String -> IO (Either ParseError a) parseFromFile p fname = do input <- readFile fname return (runP p () fname input)
