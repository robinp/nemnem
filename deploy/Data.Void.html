reasoning tool of \"ex falso quodlibet\". absurd :: Void -> a absurd a = a `seq` spin a where spin (Void b) = spin b -- | If 'Void' is uninhabited then any 'Functor' that holds only values of type vacuousM :: Monad m => m Void -> m a vacuousM = liftM absurd instance Semigroup Void where a <> _ = a times1p _ a = a instance Ix Void where range _ = [] index _ = absurd inRange _ = absurd rangeSize
