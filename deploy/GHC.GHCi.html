(e.g state monads) class (Monad m) => GHCiSandboxIO m where ghciStepIO :: m a -> IO a instance GHCiSandboxIO IO where ghciStepIO = id -- | A monad that doesn't allow any IO. newtype NoIO a = NoIO { noio :: IO a } instance Monad NoIO where return a = NoIO (return a) (>>=) k f = NoIO (noio k >>= noio . f) instance GHCiSandboxIO NoIO where ghciStepIO = noio
