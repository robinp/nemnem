import Data.Traversable (Traversable(traverse)) infixr 9 `Compose` -- | Right-to-left composition of functors. -- The composition of applicative functors is always applicative, -- but the composition of monads is not always a monad. newtype Compose f g a = Compose { getCompose :: f (g a) } -- Instances of Prelude classes -- kludge to get type with the same instances as g a newtype Apply g a =
